\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}\small
      \textbf{#1} & #2 \\

    \end{tabular*}\vspace{-7pt}
}
